- 문제 접근
    - BFS를 통해 문제를 해결, Dijkstra 처럼 방문한 정점의 Len을 최소로 유지하면서 모든 노드를 방문 

```javascript
function solution(n, edge) {
    let v = {};
    let len = Array(20001).fill(Infinity);    
    
    edge.forEach(i => {
        let f = i[0], s = i[1];
        if(v[f]) v[f].push(s);
        else v[f] = [s];
        if(v[s]) v[s].push(f);
        else v[s] = [f];
    })
    
    len[1] = 0;
    let q = [];
    q.push({
        node: 1,
        cnt: 0,
    });
    
    while(q.length){
        let now = q.shift();
        
        if(len[now.node] < now.cnt) continue;
        if(v.hasOwnProperty(now.node)){
            v[now.node].forEach(next => {
                if(len[next] > now.cnt+1){
                    len[next] = now.cnt+1;
                    q.push({
                        node:next,
                        cnt: now.cnt+1
                    })
                }
            })
        }
    }
    
    let answer = 0;
    let maxi = 0;
    for(let i = 2; i<=n; i++){
        if(maxi < len[i]){
            answer = 1;
            maxi = len[i];
        }
        else if(maxi == len[i]){
            answer++;
        }
    }
    
    return answer;
}
```
